Base Configuration

class BaseConfig(BaseModel):
    "Base configuration for all models"
    
    class Config:
        use_enum_values = True
        validate_assignment = True
        arbitrary_types_allowed = True
        json_encoders = {
            datetime: lambda v: v.isoformat(),
            Decimal: lambda v: float(v),
            UUID: lambda v: str(v)
        }

class TimestampMixin(BaseModel):
    "Mixin for timestamp fields"
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: Optional[datetime] = None
    
    @validator('updated_at', pre=True, always=True)
    def set_updated_at(cls, v):
        return v or datetime.utcnow()

class AgentMetrics(BaseConfig, TimestampMixin):
    "Core performance metrics for agents"
    
    agent_id: UUID = Field(default_factory=uuid4)
    agent_name: str = Field(..., description="Agent display name")
    agent_type: AgentType
    status: AgentStatus = AgentStatus.IDLE
    
    # Performance Metrics
    tasks_completed: int = Field(default=0, ge=0)
    tasks_failed: int = Field(default=0, ge=0)
    average_response_time: float = Field(default=0.0, ge=0.0)
    uptime_percentage: float = Field(default=100.0, ge=0.0, le=100.0)
    
    # Gaming Metrics
    experience_points: int = Field(default=0, ge=0)
    level: int = Field(default=1, ge=1)
    achievements: List[str] = Field(default_factory=list)
    
    @property
    def success_rate(self) -> float:
        "Calculate task success rate"
        total_tasks = self.tasks_completed + self.tasks_failed
        return (self.tasks_completed / total_tasks * 100) if total_tasks > 0 else 0.0
    
    @validator('level')
    def calculate_level(cls, v, values):
        "Auto-calculate level based on XP"
        xp = values.get('experience_points', 0)
        return max(1, int(xp / 1000) + 1)

class DashboardIntegration(BaseConfig):
    "Complete dashboard integration example"
    
    @staticmethod
    def create_agent_update_message(agent: AgentMetrics) -> WebSocketMessage:
        "Create real-time agent update message"
        return WebSocketMessage(
            message_type="agent_update",
            data={
                "agent_id": str(agent.agent_id),
                "agent_name": agent.agent_name,
                "status": agent.status,
                "level": agent.level,
                "xp": agent.experience_points,
                "success_rate": agent.success_rate
            }
        )
    
    @staticmethod
    def create_achievement_notification(agent_id: UUID, achievement: Achievement) -> WebSocketMessage:
        "Create achievement unlock notification"
        return WebSocketMessage(
            message_type="achievement",
            data={
                "agent_id": str(agent_id),
                "achievement_id": achievement.achievement_id,
                "achievement_name": achievement.name,
                "rarity": achievement.rarity
            },
            requires_notification=True,
            notification_title=f"Achievement Unlocked!",
            notification_body=f"{achievement.name} - {achievement.description}"
        )
